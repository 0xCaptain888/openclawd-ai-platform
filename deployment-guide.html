NVIDIA 驱动安装

安装 NVIDIA 驱动
# 方法一：使用 Ubuntu 官方仓库（推荐，最稳定） # 查看可用驱动版本 ubuntu-drivers devices # 自动安装推荐版本 sudo ubuntu-drivers autoinstall # 或手动指定版本（推荐 535 或 545 系列） sudo apt install -y nvidia-driver-545 # 重启系统 sudo reboot # 验证安装 nvidia-smi +-----------------------------------------------------------------------------------------+ | NVIDIA-SMI 545.29.06 Driver Version: 545.29.06 CUDA Version: 12.3 | | GPU Name Persistence-M | Bus-Id Disp.A | Volatile Uncorr. ECC | | 0 NVIDIA GeForce RTX 4090 On | 00000000:01:00.0 Off | Off | | 1 NVIDIA GeForce RTX 4090 On | 00000000:02:00.0 Off | Off | +-----------------------------------------------------------------------------------------+
方法二：从 NVIDIA 官网安装 .run 文件
# 禁用 Nouveau 开源驱动（必须） sudo bash -c "echo 'blacklist nouveau' >> /etc/modprobe.d/blacklist-nouveau.conf" sudo bash -c "echo 'options nouveau modeset=0' >> /etc/modprobe.d/blacklist-nouveau.conf" sudo update-initramfs -u sudo reboot # 下载并安装（从 https://www.nvidia.com/drivers 获取链接） chmod +x NVIDIA-Linux-x86_64-545.29.06.run sudo ./NVIDIA-Linux-x86_64-545.29.06.run --no-opengl-files
⚠️
常见错误处理：
• NVIDIA-SMI has failed → 驱动未正确加载，检查 Secure Boot 是否关闭
• Nouveau kernel driver in use → Nouveau 未禁用，执行上方 blacklist 步骤
• Unable to load kernel module → 内核版本与驱动不匹配，运行 sudo apt install linux-headers-$(uname -r)

nvidia-smi 监控脚本

gpu_monitor.sh
#!/bin/bash # /opt/scripts/gpu_monitor.sh # GPU 状态实时监控 + 过温告警 TEMP_THRESHOLD=85 # GPU 温度阈值（℃） MEM_THRESHOLD=95 # 显存使用率阈值（%） WEBHOOK_URL="https://qyapi.weixin.qq.com/cgi-bin/webhook/send?key=YOUR_KEY" while true; do # 获取 GPU 信息 gpu_info=$(nvidia-smi --query-gpu=index,name,temperature.gpu,utilization.gpu,memory.used,memory.total,memory.free --format=csv,noheader,nounits) echo "$gpu_info" | while IFS=',' read idx name temp util mem_used mem_total mem_free; do # 去除空格 temp=$(echo $temp | tr -d ' ') mem_used=$(echo $mem_used | tr -d ' ') mem_total=$(echo $mem_total | tr -d ' ') # 计算显存使用率 mem_pct=$((mem_used * 100 / mem_total)) # 过温告警 if [ "$temp" -ge "$TEMP_THRESHOLD" ]; then curl -s "$WEBHOOK_URL" \ -H 'Content-Type: application/json' \ -d "{\"msgtype\":\"text\",\"text\":{\"content\":\"🔥 GPU $idx ($name) 温度过高: ${temp}°C\"}}" fi # 显存告警 if [ "$mem_pct" -ge "$MEM_THRESHOLD" ]; then curl -s "$WEBHOOK_URL" \ -H 'Content-Type: application/json' \ -d "{\"msgtype\":\"text\",\"text\":{\"content\":\"⚠️ GPU $idx 显存使用率: ${mem_pct}% (${mem_used}/${mem_total} MiB)\"}}" fi echo "[$(date '+%Y-%m-%d %H:%M:%S')] GPU$idx: ${temp}°C | GPU利用率:${util}% | 显存:${mem_used}/${mem_total}MiB (${mem_pct}%)" done sleep 60 done

健康检查与告警通知

health_check.sh — 服务健康检查
#!/bin/bash # /opt/scripts/health_check.sh # 每分钟检查 AI 服务是否正常响应 SERVICE_URL="http://localhost:8000/v1/models" DINGTALK_WEBHOOK="https://oapi.dingtalk.com/robot/send?access_token=YOUR_TOKEN" MAX_RETRIES=3 check_service() { local retries=0 while [ $retries -lt $MAX_RETRIES ]; do response=$(curl -s -o /dev/null -w "%{http_code}" --max-time 10 "$SERVICE_URL") if [ "$response" = "200" ]; then return 0 fi retries=$((retries + 1)) sleep 5 done return 1 } send_alert() { local msg="$1" # 钉钉告警 curl -s "$DINGTALK_WEBHOOK" \ -H 'Content-Type: application/json' \ -d "{ \"msgtype\": \"markdown\", \"markdown\": { \"title\": \"AI服务告警\", \"text\": \"### 🚨 AI 服务异常\\n\\n$msg\\n\\n> 时间: $(date '+%Y-%m-%d %H:%M:%S')\\n> 主机: $(hostname)\" } }" # 企业微信告警 curl -s "$WECHAT_WEBHOOK" \ -H 'Content-Type: application/json' \ -d "{\"msgtype\":\"text\",\"text\":{\"content\":\"$msg - $(date)\"}}" } if ! check_service; then send_alert "vLLM 推理服务无响应，正在尝试重启..." sudo systemctl restart vllm sleep 30 if check_service; then send_alert "✅ vLLM 服务已自动恢复" else send_alert "❌ vLLM 服务重启失败，请立即人工介入！" fi fi

#!/usr/bin/env python3 # /opt/scripts/benchmark.py import asyncio, aiohttp, time, statistics API_URL = "http://localhost:8000/v1/chat/completions" CONCURRENCY_LEVELS = [1, 5, 10, 20, 50] REQUESTS_PER_LEVEL = 50 async def send_request(session): payload = { "model": "Qwen2-72B-Instruct-GPTQ-Int4", "messages": [{"role": "user", "content": "简要回答：什么是机器学习？"}], "max_tokens": 100 } start = time.perf_counter() async with session.post(API_URL, json=payload) as resp: result = await resp.json() latency = time.perf_counter() - start tokens = result["usage"]["completion_tokens"] return latency, tokens async def benchmark(concurrency): connector = aiohttp.TCPConnector(limit=concurrency) async with aiohttp.ClientSession(connector=connector) as session: sem = asyncio.Semaphore(concurrency) async def limited(): async with sem: return await send_request(session) start = time.perf_counter() results = await asyncio.gather(*[limited() for _ in range(REQUESTS_PER_LEVEL)]) wall_time = time.perf_counter() - start latencies = [r[0] for r in results] total_tokens = sum(r[1] for r in results) return { "并发数": concurrency, "QPS": round(REQUESTS_PER_LEVEL / wall_time, 1), "总吞吐(tok/s)": round(total_tokens / wall_time, 1), "P50(ms)": round(statistics.median(latencies)*1000), "P99(ms)": round(sorted(latencies)[int(len(latencies)*0.99)]*1000), } async def main(): print(f"{'并发':<6} {'QPS':<8} {'吞吐(tok/s)':<14} {'P50':<10} {'P99':<10}") print("-" * 50) for c in CONCURRENCY_LEVELS: r = await benchmark(c) print(f"{r['并发数']:<6} {r['QPS']:<8} {r['总吞吐(tok/s)']:<14} {r['P50(ms)']:<10} {r['P99(ms)']:<10}") asyncio.run(main())

t < 60] if len(state["minute"]) >= config["rpm"]: raise HTTPException(429, f"Rate limit: {config['rpm']} req/min") # 每日限额 if state["daily"] >= config["daily"]: raise HTTPException(429, "Daily quota exceeded") state["minute"].append(now) state["daily"] += 1 # ===== 审计日志 ===== audit_logger = logging.getLogger("audit") handler = logging.FileHandler("/var/log/ai-gateway/audit.jsonl") audit_logger.addHandler(handler) audit_logger.setLevel(logging.INFO) # ===== 鉴权依赖 ===== async def verify_key(credentials: HTTPAuthorizationCredentials = Depends(security)): key = credentials.credentials if key not in API_KEYS: raise HTTPException(401, "Invalid API key") check_rate_limit(key) return key # ===== 代理转发 ===== @app.api_route("/v1/{path:path}", methods=["GET","POST"]) async def proxy(request: Request, path: str, api_key: str = Depends(verify_key)): start = time.time() body = await request.body() # 转发到 vLLM async with httpx.AsyncClient(timeout=300) as client: resp = await client.request( method=request.method, url=f"{VLLM_BACKEND}/v1/{path}", content=body, headers={"Content-Type": "application/json"} ) latency = time.time() - start # 记录审计日志 audit_logger.info(json.dumps({ "ts": datetime.now().isoformat(), "key": api_key[:10] + "...", "client": API_KEYS[api_key]["name"], "path": path, "status": resp.status_code, "latency_ms": round(latency * 1000), "ip": request.client.host, })) return resp.json()

#!/usr/bin/env python3 # /opt/eval/run_eval.py import json, sys, re from openai import OpenAI from datetime import datetime API_URL = sys.argv[1] if len(sys.argv) > 1 else "http://localhost:8000/v1" MODEL = sys.argv[2] if len(sys.argv) > 2 else "auto" client = OpenAI(base_url=API_URL, api_key="local") test_cases = json.load(open("test_cases.json")) results = [] passed = 0 total = len(test_cases) for tc in test_cases: print(f" 测试 {tc['id']}...", end=" ") messages = [{"role": "user", "content": tc["prompt"]}] if "context" in tc: messages.insert(0, {"role": "system", "content": f"根据以下信息回答：{tc['context']}"}) resp = client.chat.completions.create( model=MODEL, messages=messages, max_tokens=500, temperature=0.3 ) answer = resp.choices[0].message.content criteria = tc["criteria"] checks = [] # ===== 规则检查 ===== if "min_length" in criteria: checks.append(("长度≥最小", len(answer) >= criteria["min_length"])) if "max_length" in criteria: checks.append(("长度≤最大", len(answer) <= criteria["max_length"])) if "must_contain" in criteria: for kw in criteria["must_contain"]: checks.append((f"包含'{kw}'", kw in answer)) if "must_not_contain" in criteria: for kw in criteria["must_not_contain"]: checks.append((f"不含'{kw}'", kw not in answer)) if criteria.get("must_refuse"): refuse_kws = ["抱歉", "无法", "不能", "违法", "不当"] checks.append(("正确拒绝", any(k in answer for k in refuse_kws))) test_passed = all(c[1] for c in checks) if test_passed: passed += 1 status = "✅ PASS" if test_passed else "❌ FAIL" print(status) failed_checks = [c[0] for c in checks if not c[1]] results.append({ "id": tc["id"], "category": tc["category"], "passed": test_passed, "failed_checks": failed_checks, "answer_preview": answer[:100] }) # ===== 汇总 ===== score = passed / total * 100 print(f"\n{'='*50}") print(f"评估结果: {passed}/{total} 通过 ({score:.1f}%)") print(f"质量门禁: {'✅ 通过' if score >= 90 else '❌ 未通过（需 ≥90%）'}") # 保存结果 report = { "timestamp": datetime.now().isoformat(), "model": MODEL, "api_url": API_URL, "score": score, "passed": passed, "total": total, "gate_passed": score >= 90, "details": results } with open(f"eval_results_{datetime.now():%Y%m%d_%H%M}.json", "w") as f: json.dump(report, f, ensure_ascii=False, indent=2) sys.exit(0 if score >= 90 else 1)
